246 #define PIP_TASKQ_INIT(L) \
247  do { PIP_TASKQ_NEXT(L) = (L); PIP_TASKQ_PREV(L) = (L); } while(0)
248 
249 #define PIP_TASKQ_NEXT(L) ((L)->next)
250 #define PIP_TASKQ_PREV(L) ((L)->prev)
251 #define PIP_TASKQ_PREV_NEXT(L) ((L)->prev->next)
252 #define PIP_TASKQ_NEXT_PREV(L) ((L)->next->prev)
253 
254 #define PIP_TASKQ_ENQ_FIRST(L,E) \
255  do { PIP_TASKQ_NEXT(E) = PIP_TASKQ_NEXT(L); \
256  PIP_TASKQ_PREV(E) = (L); \
257  PIP_TASKQ_NEXT_PREV(L) = (E); \
258  PIP_TASKQ_NEXT(L) = (E); } while(0)
259 
260 #define PIP_TASKQ_ENQ_LAST(L,E) \
261  do { PIP_TASKQ_NEXT(E) = (L); \
262  PIP_TASKQ_PREV(E) = PIP_TASKQ_PREV(L); \
263  PIP_TASKQ_PREV_NEXT(L) = (E); \
264  PIP_TASKQ_PREV(L) = (E); } while(0)
265 
266 #define PIP_TASKQ_DEQ(L) \
267  do { PIP_TASKQ_NEXT_PREV(L) = PIP_TASKQ_PREV(L); \
268  PIP_TASKQ_PREV_NEXT(L) = PIP_TASKQ_NEXT(L); \
269  PIP_TASKQ_INIT(L); } while(0)
270 
271 #define PIP_TASKQ_MOVE_QUEUE(P,Q) \
272  do { if( !PIP_TASKQ_ISEMPTY(Q) ) { \
273  PIP_TASKQ_NEXT_PREV(Q) = (P); \
274  PIP_TASKQ_PREV_NEXT(Q) = (P); \
275  PIP_TASKQ_NEXT(P) = PIP_TASKQ_NEXT(Q); \
276  PIP_TASKQ_PREV(P) = PIP_TASKQ_PREV(Q); \
277  PIP_TASKQ_INIT(Q); } } while(0)
278 
279 #define PIP_TASKQ_ISEMPTY(L) \
280  ( PIP_TASKQ_NEXT(L) == (L) && PIP_TASKQ_PREV(L) == (L) )
281 
282 #define PIP_TASKQ_FOREACH(L,E) \
283  for( (E)=(L)->next; (L)!=(E); (E)=PIP_TASKQ_NEXT(E) )
284 
285 #define PIP_TASKQ_FOREACH_SAFE(L,E,TV) \
286  for( (E)=(L)->next, (TV)=PIP_TASKQ_NEXT(E); \
287  (L)!=(E); \
288  (E)=(TV), (TV)=PIP_TASKQ_NEXT(TV) )
