88 #define PIP_TASKQ_NEXT(L) (((pip_task_t*)(L))->next)
89 #define PIP_TASKQ_PREV(L) (((pip_task_t*)(L))->prev)
90 #define PIP_TASKQ_PREV_NEXT(L) (((pip_task_t*)(L))->prev->next)
91 #define PIP_TASKQ_NEXT_PREV(L) (((pip_task_t*)(L))->next->prev)

93 #ifdef AH
94 #define PIP_TASKQ_CHECK(Q) \
95  ASSERTD( PIP_TASKQ_NEXT(Q) != PIP_TASKQ_PREV(Q) )
96 #else
97 #define PIP_TASKQ_CHECK(Q)
98 #endif
99 
100 #define PIP_TASKQ_INIT(L) \
101  do { PIP_TASKQ_NEXT(L) = PIP_TASKQ_PREV(L) = \
102  (pip_task_t*)(L); } while(0)
103 
104 #define PIP_TASKQ_ENQ_FIRST(L,E) \
105  do { PIP_TASKQ_CHECK(E); \
106  PIP_TASKQ_NEXT(E) = PIP_TASKQ_NEXT(L); \
107  PIP_TASKQ_PREV(E) = (pip_taskt*)(L); \
108  PIP_TASKQ_NEXT_PREV(L) = (pip_task_t*)(E); \
109  PIP_TASKQ_NEXT(L) = (pip_task_t*)(E); } while(0)
110 
111 #define PIP_TASKQ_ENQ_LAST(L,E) \
112  do { PIP_TASKQ_CHECK(E); \
113  PIP_TASKQ_NEXT(E) = (pip_task_t*)(L); \
114  PIP_TASKQ_PREV(E) = PIP_TASKQ_PREV(L); \
115  PIP_TASKQ_PREV_NEXT(L) = (pip_task_t*)(E); \
116  PIP_TASKQ_PREV(L) = (pip_task_t*)(E); } while(0)
117 
118 #define PIP_TASKQ_DEQ(E) \
119  do { PIP_TASKQ_NEXT_PREV(E) = PIP_TASKQ_PREV(E); \
120  PIP_TASKQ_PREV_NEXT(E) = PIP_TASKQ_NEXT(E); \
121  PIP_TASKQ_INIT(E); } while(0)
122 
123 #define PIP_TASKQ_APPEND(P,Q) \
124  do { if( !PIP_TASKQ_ISEMPTY(Q) ) { \
125  PIP_TASKQ_NEXT_PREV(Q) = PIP_TASKQ_PREV(P); \
126  PIP_TASKQ_PREV_NEXT(Q) = (P); \
127  PIP_TASKQ_PREV_NEXT(P) = PIP_TASKQ_NEXT(Q); \
128  PIP_TASKQ_PREV(P) = PIP_TASKQ_PREV(Q); \
129  PIP_TASKQ_INIT(Q); } } while(0)
130 
131 #define PIP_TASKQ_ISEMPTY(L) \
132  ( PIP_TASKQ_NEXT(L) == (L) && PIP_TASKQ_PREV(L) == (pip_task_t*)(L) )
133 
134 #define PIP_TASKQ_FOREACH(L,E) \
135  for( (E)=PIP_TASKQ_NEXT(L); (pip_task_t*)(L)!=(E); \
136  (E)=PIP_TASKQ_NEXT(E) )
137 
138 #define PIP_TASKQ_FOREACH_SAFE(L,E,TV) \
139  for( (E)=PIP_TASKQ_NEXT(L), (TV)=PIP_TASKQ_NEXT(E); \
140  (pip_task_t*)(L)!=(E); \
141  (E)=(TV), (TV)=PIP_TASKQ_NEXT(TV) )
142 
143 #define PIP_TASKQ_MOVE(D,S) \
144  do { if( PIP_TASKQ_ISEMPTY(S) ) { \
145  PIP_TASKQ_INIT(D); \
146  } else { \
147  (D)->next = (S)->next; \
148  (D)->prev = (S)->prev; \
149  PIP_TASKQ_INIT(S); } } while(0)

569  fprintf( fp, "%s: Queue[%d/%d]:%s\n", tag, i++, queue->length, msg );
570  }
571  }
572  } else {
573  queue->methods->describe( (void*) queue, fp );
574  }
575  }
