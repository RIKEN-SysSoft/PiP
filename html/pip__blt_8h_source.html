97 #define PIP_TASKQ_INIT(L) \
98  do { PIP_TASKQ_NEXT(L) = PIP_TASKQ_PREV(L) = \
99  (pip_task_t*)(L); } while(0)
100 
101 #define PIP_TASKQ_ENQ_FIRST(L,E) \
102  do { PIP_TASKQ_CHECK(E); \
103  PIP_TASKQ_NEXT(E) = PIP_TASKQ_NEXT(L); \
104  PIP_TASKQ_PREV(E) = (pip_taskt*)(L); \
105  PIP_TASKQ_NEXT_PREV(L) = (pip_task_t*)(E); \
106  PIP_TASKQ_NEXT(L) = (pip_task_t*)(E); } while(0)
107 
108 #define PIP_TASKQ_ENQ_LAST(L,E) \
109  do { PIP_TASKQ_CHECK(E); \
110  PIP_TASKQ_NEXT(E) = (pip_task_t*)(L); \
111  PIP_TASKQ_PREV(E) = PIP_TASKQ_PREV(L); \
112  PIP_TASKQ_PREV_NEXT(L) = (pip_task_t*)(E); \
113  PIP_TASKQ_PREV(L) = (pip_task_t*)(E); } while(0)
114 
115 #define PIP_TASKQ_DEQ(E) \
116  do { PIP_TASKQ_NEXT_PREV(E) = PIP_TASKQ_PREV(E); \
117  PIP_TASKQ_PREV_NEXT(E) = PIP_TASKQ_NEXT(E); \
118  PIP_TASKQ_INIT(E); } while(0)
119 
120 #define PIP_TASKQ_APPEND(P,Q) \
121  do { if( !PIP_TASKQ_ISEMPTY(Q) ) { \
122  PIP_TASKQ_NEXT_PREV(Q) = PIP_TASKQ_PREV(P); \
123  PIP_TASKQ_PREV_NEXT(Q) = (P); \
124  PIP_TASKQ_PREV_NEXT(P) = PIP_TASKQ_NEXT(Q); \
125  PIP_TASKQ_PREV(P) = PIP_TASKQ_PREV(Q); \
126  PIP_TASKQ_INIT(Q); } } while(0)
127 
128 #define PIP_TASKQ_ISEMPTY(L) \
129  ( PIP_TASKQ_NEXT(L) == (L) && PIP_TASKQ_PREV(L) == (pip_task_t*)(L) )
130 
131 #define PIP_TASKQ_FOREACH(L,E) \
132  for( (E)=PIP_TASKQ_NEXT(L); (pip_task_t*)(L)!=(E); \
133  (E)=PIP_TASKQ_NEXT(E) )
134 
135 #define PIP_TASKQ_FOREACH_SAFE(L,E,TV) \
136  for( (E)=PIP_TASKQ_NEXT(L), (TV)=PIP_TASKQ_NEXT(E); \
137  (pip_task_t*)(L)!=(E); \
138  (E)=(TV), (TV)=PIP_TASKQ_NEXT(TV) )
139 
140 #define PIP_TASKQ_MOVE(D,S) \
141  do { if( PIP_TASKQ_ISEMPTY(S) ) { \
142  PIP_TASKQ_INIT(D); \
143  } else { \
144  (D)->next = (S)->next; \
145  (D)->prev = (S)->prev; \
146  PIP_TASKQ_INIT(S); } } while(0)
