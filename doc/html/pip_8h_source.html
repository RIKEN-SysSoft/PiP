334 void pip_spawn_from_main( pip_spawn_program_t *progp,
335  char *prog, char **argv, char **envv,
336  void *exp, void *aux ) {
337  memset( progp, 0, sizeof(pip_spawn_program_t) );
338  if( prog != NULL ) {
339  progp->prog = prog;
340  } else {
341  progp->prog = argv[0];
342  }
343  progp->argv = argv;
344  if( envv == NULL ) {
345  extern char **environ;
346  progp->envv = environ;
347  } else {
348  progp->envv = envv;
349  }
350  progp->exp = exp;
351  progp->aux = aux;
352 }
