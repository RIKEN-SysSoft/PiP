314 void pip_spawn_from_main( pip_spawn_program_t *progp,
315  char *prog, char **argv, char **envv,
316  void *exp ) {
317  memset( progp, 0, sizeof(pip_spawn_program_t) );
318  if( prog != NULL ) {
319  progp->prog = prog;
320  } else {
321  progp->prog = argv[0];
322  }
323  progp->argv = argv;
324  if( envv == NULL ) {
325  extern char **environ;
326  progp->envv = environ;
327  } else {
328  progp->envv = envv;
329  }
330  progp->exp = exp;
331 }
