85 #define PIP_TASKQ_CHECK(Q) \
86  ASSERTD( PIP_TASKQ_NEXT(Q) != PIP_TASKQ_PREV(Q) )
87 #else
88 #define PIP_TASKQ_CHECK(Q)
89 #endif

91 #define PIP_TASKQ_INIT(L) \
92  do { PIP_TASKQ_NEXT(L) = PIP_TASKQ_PREV(L) = \
93  (pip_task_t*)(L); } while(0)
94 
95 #define PIP_TASKQ_ENQ_FIRST(L,E) \
96  do { PIP_TASKQ_CHECK(E); \
97  PIP_TASKQ_NEXT(E) = PIP_TASKQ_NEXT(L); \
98  PIP_TASKQ_PREV(E) = (pip_taskt*)(L); \
99  PIP_TASKQ_NEXT_PREV(L) = (pip_task_t*)(E); \
100  PIP_TASKQ_NEXT(L) = (pip_task_t*)(E); } while(0)
101 
102 #define PIP_TASKQ_ENQ_LAST(L,E) \
103  do { PIP_TASKQ_CHECK(E); \
104  PIP_TASKQ_NEXT(E) = (pip_task_t*)(L); \
105  PIP_TASKQ_PREV(E) = PIP_TASKQ_PREV(L); \
106  PIP_TASKQ_PREV_NEXT(L) = (pip_task_t*)(E); \
107  PIP_TASKQ_PREV(L) = (pip_task_t*)(E); } while(0)
108 
109 #define PIP_TASKQ_DEQ(E) \
110  do { PIP_TASKQ_NEXT_PREV(E) = PIP_TASKQ_PREV(E); \
111  PIP_TASKQ_PREV_NEXT(E) = PIP_TASKQ_NEXT(E); \
112  PIP_TASKQ_INIT(E); } while(0)
113 
114 #define PIP_TASKQ_APPEND(P,Q) \
115  do { if( !PIP_TASKQ_ISEMPTY(Q) ) { \
116  PIP_TASKQ_NEXT_PREV(Q) = PIP_TASKQ_PREV(P); \
117  PIP_TASKQ_PREV_NEXT(Q) = (P); \
118  PIP_TASKQ_PREV_NEXT(P) = PIP_TASKQ_NEXT(Q); \
119  PIP_TASKQ_PREV(P) = PIP_TASKQ_PREV(Q); \
120  PIP_TASKQ_INIT(Q); } } while(0)
121 
122 #define PIP_TASKQ_ISEMPTY(L) \
123  ( PIP_TASKQ_NEXT(L) == (L) && PIP_TASKQ_PREV(L) == (pip_task_t*)(L) )
124 
125 #define PIP_TASKQ_FOREACH(L,E) \
126  for( (E)=PIP_TASKQ_NEXT(L); (pip_task_t*)(L)!=(E); \
127  (E)=PIP_TASKQ_NEXT(E) )
128 
129 #define PIP_TASKQ_FOREACH_SAFE(L,E,TV) \
130  for( (E)=PIP_TASKQ_NEXT(L), (TV)=PIP_TASKQ_NEXT(E); \
131  (pip_task_t*)(L)!=(E); \
132  (E)=(TV), (TV)=PIP_TASKQ_NEXT(TV) )
133 
134 #define PIP_TASKQ_MOVE(D,S) \
135  do { if( PIP_TASKQ_ISEMPTY(S) ) { \
136  PIP_TASKQ_INIT(D); \
137  } else { \
138  (D)->next = (S)->next; \
139  (D)->prev = (S)->prev; \
140  PIP_TASKQ_INIT(S); } } while(0)
